Подсчет количества дней

function getDays(day) {
    const date = new Date(day);
    const now = Date.now();
    let remaining = date - now;
    remaining /= 1000; // секунды до даты
    remaining /= 60;    // минуты до даты
    remaining /= 60;    // часы до даты
    remaining /= 24;
    return remaining;
}
